code Mã nguồn thuật toán (C++) 

void kruskal(int V, vector<Edge>& edges) {
    sort(edges.begin(), edges.end());
    DisjointSet dsu(V);
    vector<Edge> mst;

    for (auto& edge : edges) {
        if (dsu.unite(edge.u, edge.v)) {
            mst.push_back(edge);
            if (mst.size() == V - 1) break;
        }
    }
}
