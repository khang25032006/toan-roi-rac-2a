def floyd_warshall(matrix, V):
    D = [[cell for cell in row] for row in matrix]
    P = [[i for j in range(V)] for i in range(V)]
    
    for k in range(V):
        for i in range(V):
            for j in range(V):
                if D[i][k] + D[k][j] < D[i][j]:
                    D[i][j] = D[i][k] + D[k][j]
                    P[i][j] = P[k][j]
                    
    return D, P
