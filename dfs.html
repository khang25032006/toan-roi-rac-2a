def dfs_de_quy(graph, u, visited, dfs_order):
    visited[u] = True
    dfs_order.append(u)
    for v in range(1, len(graph) + 1):
        if graph[u-1][v-1] == 1 and not visited[v]:
            dfs_de_quy(graph, v, visited, dfs_order)
