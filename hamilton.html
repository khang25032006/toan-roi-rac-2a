def hamiltonian_cycle(graph, path, pos):
    if pos == len(graph):
        return graph[path[pos-1]][path[0]] == 1
    for v in range(len(graph)):
        if is_valid(v, graph, path, pos):
            path[pos] = v
            if hamiltonian_cycle(graph, path, pos + 1):
                return True
            path[pos] = -1
    return False
