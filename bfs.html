code Mã nguồn thuật toán (Python) 

def bfs(graph, start):
    visited = [False] * (len(graph) + 1)
    queue = [start]
    visited[start] = True
    bfs_order = []
    
    while queue:
        u = queue.pop(0)
        bfs_order.append(u)
        for v in range(1, len(graph) + 1):
            if graph[u-1][v-1] == 1 and not visited[v]:
                queue.append(v)
                visited[v] = True
    return bfs_order
